# 按步长2取列表数据
>>> a = [1,2,3,4,5]
>>> a[::2] 
[1, 3, 5]

# 用步长-1来反转列表
>>> a[::-1]
[5, 4, 3, 2, 1]

# 用切片来删除列表的某一段
>>> a = [1,2,3,4,5]
>>> a[1:3] = []
>>> a
[1, 4, 5]

# 也可以用 del a[1:3]
>>> a = [1,2,3,4,5]
>>> del a[1:3]
>>> a
[1, 4, 5]

更安全的做法：

 1

 3

>>> def foo(x=None):
...     if x is None:
...         x = []
...     x.append(1)
...     print x
...
>>> foo()

>>> foo()

>>>
